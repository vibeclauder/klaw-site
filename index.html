# Multi-factor momentum strategy core

import pandas as pd
import numpy as np

class MomentumStrategy:
    """Scores instruments on price momentum,
    volatility-adjusted return, and
    cross-sectional relative strength."""

    def score(self, df: pd.DataFrame) -> float:
        # 12-1 month price momentum (Jegadeesh)
        mom = df['close'].pct_change(self.lookback)

        # Volatility-normalize the return signal
        vol = (df['close']
               .pct_change()
               .rolling(21)
               .std() * np.sqrt(252))

        return (mom / vol).rank(pct=True)

    def build_portfolio(self, scores, capital):
        top = scores.nlargest(self.n_positions)
        weights = top / top.sum()
        return {
            sym: capital * w
            for sym, w in weights.items()
        }

# 14-point pre-trade risk gate

class RiskGate:
    CHECKS = [
        'portfolio_heat',    # max open risk %
        'position_size',     # per-trade limit
        'sector_exposure',   # sector cap
        'correlation_limit', # corr matrix
        'drawdown_gate',     # DD circuit break
        'vol_regime',        # VIX filter
        'liquidity_check',   # ADV threshold
        'spread_check',      # bid/ask limit
    ]

    def clear(self, order) -> bool:
        for check in self.CHECKS:
            if not getattr(self, check)(order):
                self._log_block(check, order)
                return False
        return True

    def portfolio_heat(self, order) -> bool:
        current = self.portfolio.open_risk()
        return current + order.risk <= self.max_heat
